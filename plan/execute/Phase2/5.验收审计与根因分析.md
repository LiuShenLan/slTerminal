#### 4.1.2 预检阶段

**主 worktree 检测**（方式 B——规范化路径比较）：

```rust
fn validate_worktree_is_not_main(repo_path: &str, worktree_path: &str) -> Result<(), AppError> {
    let repo = git2::Repository::open(repo_path)
        .map_err(|e| AppError::Git(format!("无法打开仓库: {e}")))?;
    let main_path = repo.workdir()
        .ok_or_else(|| AppError::Git("仓库无工作目录".to_string()))?;
    let worktree_canon = Path::new(worktree_path).canonicalize()
        .unwrap_or_else(|_| PathBuf::from(worktree_path));
    let main_canon = main_path.canonicalize()
        .unwrap_or_else(|_| main_path.to_path_buf());
    if worktree_canon == main_canon {
        return Err(AppError::Worktree("不能删除主工作树".to_string()));
    }
    Ok(())
}
```

**脏状态检测**（新增 `git_worktree_status` 命令，分三级）：

| 级别 | 条件 | 处理 |
|------|------|------|
| Clean | 无任何更改 | 直接允许 |
| Dirty | 仅有 unstaged/untracked 更改 | 警告 + checkbox gate |
| Blocked | 进行中 merge/rebase/cherry-pick | 阻止删除 |

**活跃终端会话 + 关联页面**：删除前通过 `getSessionsByWorktree(worktreePath)` 获取受影响会话，并行 `pty.kill`。

#### 4.1.3 后端：三级回退删除序列

libgit2 无 `git_worktree_remove` API——唯一可靠方式通过 shell 调用。

```
Level 1: git worktree remove --force <path>     (优雅: 清理元数据 + 目录)
  ├─ exit 0              → prune
  ├─ "not a working tree" → Level 2
  ├─ "file locked"        → 重试(最多2次, 500ms)
  └─ 其他错误             → 返回错误

Level 2: cmd.exe /c (attrib -R + rmdir /S /Q)  (安全: 不穿透 NTFS junction)
  ├─ 成功 → prune
  └─ 失败 → Level 3

Level 3: remove_dir_all crate v0.8             (健壮: 只读/交接点/长路径)
          + 指数退避(50ms→2s, max 10s, ±50% jitter)
  ├─ 成功 → prune
  └─ 失败 → 返回最终错误

prune: git worktree prune                       (清理元数据残留)
```

**Windows 关键纠正**：`ERROR_SHARING_VIOLATION`(32)/`ERROR_LOCK_VIOLATION`(33) 在 Rust 中映射为 `ErrorKind::Other`，必须通过 `err.raw_os_error()` 检测（非 `err.kind()`）。

// git/worktree.rs — 接受完整路径（非分支名）
#[tauri::command]
pub async fn git_worktree_remove(
    repo_path: String,
    worktree_path: String,  // ★ 不再从分支名重建
) -> Result<WorktreeRemoveResult, AppError> {
    validate_worktree_is_not_main(&repo_path, &worktree_path)?;

    // Level 1: git worktree remove --force
    let _ = try_git_worktree_remove(&repo_path, &worktree_path, 2, 500);

    // Level 2: cmd rmdir
    if Path::new(&worktree_path).exists() {
        try_cmd_rmdir(&worktree_path)?;
    }

    // Level 3: remove_dir_all crate + 指数退避
    if Path::new(&worktree_path).exists() {
        let path = format!("\\\\?\\{}", worktree_path);
        tokio::task::spawn_blocking(move || remove_dir_all_robust(&path)).await??;
    }

    if Path::new(&worktree_path).exists() {
        return Err(AppError::Worktree("删除失败: 目录仍存在, 可能被占用".to_string()));
    }

    let _ = Command::new("git").args(["worktree", "prune"]).current_dir(&repo_path).output();
    Ok(WorktreeRemoveResult { success: true, path: worktree_path })
}

#### 4.1.6 前端完整删除函数

```typescript
// src/features/worktree/useWorktreeDelete.ts（新建）
export async function deleteWorktree(params: {
  projectId: string; repoPath: string; worktreePath: string;
}): Promise<void> {
  const { projectId, repoPath, worktreePath } = params;
  const store = useProjects.getState();
  const project = store.projects[projectId];
  if (!project) throw new Error("项目不存在");

  // 1. Kill 受影响终端会话
  const sessions = getSessionsByWorktree(worktreePath);
  if (sessions.length > 0) {
    await Promise.all(sessions.map(s => pty.kill(s.sessionId)));
  }

  // 2. Store: 移除 worktree + 绑定页面
  store.removeWorktree(projectId, worktreePath);
  for (const page of project.pages.filter(p => p.binding?.worktreePath === worktreePath)) {
    store.removePage(projectId, page.pageId);
  }

  // 3. 无剩余页面时创建默认空页面
  const updated = useProjects.getState().projects[projectId];
  if (updated && updated.pages.length === 0) {
    store.addPage(projectId, { pageId: createPageId(), name: "默认",
      layout: {}, createdAt: Date.now(), lastAccessedAt: Date.now() });
  }

  // 4. 后端清理（三级回退）
  await git.removeWorktree(repoPath, worktreePath);
}
```

**新建文件**：`src/panels/terminal/TerminalRegistry.ts`

```typescript
// src/panels/terminal/TerminalRegistry.ts
// 模块级 Map，脱离 React 渲染循环 —— 终端实例在面板卸载后仍存活

interface PtySessionEntry {
  sessionId: string;
  panelId: string;
  cwd?: string;
  createdAt: number;
}

const registry = new Map<string, PtySessionEntry>();

export const TerminalRegistry = {
  /** spawn 成功后注册 */
  register(panelId: string, sessionId: string, cwd?: string): void {
    registry.set(panelId, { sessionId, panelId, cwd, createdAt: Date.now() });
  },

  /** 面板挂载时查询有无存活 session */
  get(panelId: string): PtySessionEntry | undefined {
    return registry.get(panelId);
  },

  /** 用户主动关闭面板 — 杀 PTY + 注销 */
  async destroy(panelId: string): Promise<void> {
    const entry = registry.get(panelId);
    if (entry) {
      try { await import("../../ipc/pty").then(m => m.pty.kill(entry!.sessionId)); } catch {}
      registry.delete(panelId);
    }
  },

  /** 删除 worktree 时级联清理所有关联 session */
  getByWorktree(worktreePath: string): PtySessionEntry[] {
    const results: PtySessionEntry[] = [];
    for (const [, entry] of registry) {
      if (entry.cwd?.startsWith(worktreePath)) {
        results.push(entry);
      }
    }
    return results;
  },

  /** 获取所有已注册 panelId（用于 GC） */
  getAll(): Map<string, PtySessionEntry> {
    return registry;
  },
};
```
